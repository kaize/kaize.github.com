### ~/.gitconfig

{% highlight bash linenos %}

    [color]
      diff = auto
      status = auto
      branch = auto
      interactive = auto
      ui = true
      pager = true

    [color "branch"]
      current = yellow reverse
      local = yellow
      remote = green

    [color "diff"]
      meta = yellow bold
      frag = magenta bold
      old = red bold
      new = green bold

    [color "status"]
      added = yellow
      changed = green
      untracked = cyan
    [alias]
      co = checkout
      br = branch
      ci = commit
      st = status
        unstage = reset HEAD --
        last = log -1 HEAD
        up = pull --rebase
        put = push origin HEAD

{% endhighlight %}

    # gem install capistrano_colors

    def notify(message, body, urgency, icon = :info)
      system("notify-send --urgency=#{urgency} --icon=#{icon} '#{message}' '#{body}'")
    end

    on :exit do
      notify('Capistrano Task: Finished', ARGV.join(' '), :low)
    end

    require 'capistrano_colors'
